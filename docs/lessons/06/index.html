# Code a Square Spirograph

# Step 1: create a square

main.py

from turtle import *

my_turtle = Turtle()

def rect():
    for i in range(4):
        my_turtle.forward(50)
        my_turtle.left(90)
      
rect()

# Step 5: Coloring

main.py













 
 



from turtle import *

my_turtle = Turtle()
my_turtle.speed(10)
my_turtle.pensize(1)


def rect():
    for i in range(4):
        my_turtle.forward(50)
        my_turtle.left(90)
      
for each_color in ["red","#148F77","blue","#FFC300","black"]:
    my_turtle.color(each_color)
    rect()
    my_turtle.left(12)
